class NewsSpider(scrapy.Spider):
    name = 'news_spider'
    
    def __init__(self, source_name, *args, **kwargs):
        super().__init__(*args, **kwargs)
        self.source_name = source_name
        self.start_urls = []
    
    def parse(self, response):
        # 解析列表页
        for article in response.css('article'):
            yield {
                'title': article.css('h2::text').get(),
                'url': article.css('a::attr(href)').get(),
                'published_at': article.css('time::text').get(),
            }
        
        # 跟进下一页
        next_page = response.css('a.next::attr(href)').get()
        if next_page:
            yield response.follow(next_page, self.parse)
```

# processors/rss_parser.py
import feedparser

class RSSParser:
    def __init__(self, source_config):
        self.config = source_config
    
    def fetch(self):
        feed = feedparser.parse(self.config['url'])
        articles = []
        
        for entry in feed.entries:
            articles.append({
                'title': entry.get('title', ''),
                'content': entry.get('summary', ''),
                'url': entry.get('link', ''),
                'published_at': self._parse_date(entry),
                'source_name': self.config['name'],
            })
        
        return articles
    
    def _parse_date(self, entry):
        if hasattr(entry, 'published_parsed'):
            return datetime(*entry.published_parsed[:6])
        return datetime.now()
```

### 4.2 数据清洗模块

#### 4.2.1 SimHash去重

```python
# processors/deduplicator.py
import rehash
from simhash import Simhash

class Deduplicator:
    def __init__(self):
        self.hash_index = {}
    
    def is_duplicate(self, content, threshold=3):
        """检查内容是否重复
        threshold: 哈希位数差异阈值
        """
        if not content:
            return True
        
        content = self._preprocess(content)
        hash_value = Simhash(content).value
        
        for stored_hash in self.hash_index.values():
            distance = Simhash.distance(
                Simhash(value=hash_value), 
                Simhash(value=stored_hash)
            )
            if distance <= threshold:
                return True
        
        self.hash_index[hash_value] = hash_value
        return False
    
    def _preprocess(self, text):
        # 预处理：去除特殊字符、转小写
        text = re.sub(r'[^\w\s]', '', text)
        return text.lower()
```

from ltp import LTP

        segment, hidden = self.ltp.seg([text])
        ner = self.ltp.ner(hidden)
        
        companies = []
        for item in ner[0]:
            if item[1] == 'Nh':  # 人名
                pass
            elif item[1] == 'Ni':  # 机构名
                companies.append(item[2])
        
        return companies
    
    def classify_category(self, text):
        """文章分类"""
        keywords = self.extract_keywords(text)
        # 简单规则分类

### 4.4 线索提取模块

#### 4.4.1 线索识别器

```python
# processors/lead_extractor.py
import re
from datetime import datetime

class LeadExtractor:
    def __init__(self, rules):
        self.rules = rules
    
    def extract(self, article):
        """从文章中提取线索"""
        leads = []
        text = article['title'] + ' ' + article.get('content', '')
        
        for rule in self.rules:
            if not rule['enabled']:
                continue
            
            for pattern in rule['patterns']:
                matches = re.finditer(pattern, text, re.IGNORECASE)
                for match in matches:
                    lead = {
                        'company_name': self._extract_company(text, match),
                        'event_type': rule['event_type'],
                        'event_detail': match.group(0),
                        'source_article_id': article['id'],
                        'source_title': article['title'],
                        'source_url': article['url'],
                        'source_name': article['source_name'],
                        'published_at': article.get('published_at'),
                        'confidence': self._calculate_confidence(match, rule),
                    }
                    leads.append(lead)
        
        return leads
    
    def _extract_company(self, text, match):
        """提取企业名称"""
        # 简单实现：从匹配位置前后查找公司名称模式
        context = text[max(0, match.start()-50):match.end()+50]
        
        # 匹配 "X公司"、"X科技" 等模式
        company_match = re.search(r'([^\s]{2,15}(?:公司|科技|有限|集团|企业))', context)
        if company_match:
            return company_match.group(1)
        
        return '未知'
    
    def _calculate_confidence(self, match, rule):
        """计算置信度"""
        base_score = 60
        base_score += rule.get('priority', 5) * 3
        
        # 根据匹配长度调整
        if len(match.group(0)) > 20:
            base_score += 10
        
        return min(100, base_score)
```

      retries: 5

  redis:
    image: redis:7-alpine
    container_name: leadmine_redis
    ports:
      - "6379:6379"
    volumes:
      - redis_data:/data

  rsshub:
    image: diygod/rsshub:latest
    container_name: leadmine_rsshub
    ports:
      - "1200:1200"
    environment:
      - PORT=1200
      - NODE_ENV=production

  backend:
    build:
      context: ./backend
      dockerfile: Dockerfile
    container_name: leadmine_backend
    ports:
      - "8000:8000"
    environment:
      - DATABASE_URL=mysql+pymysql://${MYSQL_USER}:${MYSQL_PASSWORD}@mysql:3306/leadmine
      - REDIS_URL=redis://redis:6379
      - ELASTICSEARCH_URL=http://elasticsearch:9200
    depends_on:
      mysql:
        condition: service_healthy
      redis:
        condition: service_started
      elasticsearch:
        condition: service_healthy
    volumes:
      - ./backend:/app

  frontend:
    build:
      context: ./web
      dockerfile: Dockerfile
    container_name: leadmine_frontend
    ports:
      - "8501:80"
    depends_on:
      - backend

volumes:
  mysql_data:

  redis_data:

networks:
  default:
    name: leadmine_network
